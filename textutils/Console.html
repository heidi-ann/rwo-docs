type color = [ `Black | `Blue | `Cyan | `Green | `Magenta | `Red | `White | `Yellow ] 
type attr = [ `Bg of color
| `Black
| `Blue
| `Bright
| `Cyan
| `Dim
| `Green
| `Magenta
| `Red
| `Reverse
| `Underscore
| `White
| `Yellow ]

val is_color_tty : unit -> bool
val width : unit -> [ `Cols of int | `Not_a_tty | `Not_available ]
The width in characters of the current output. Returns None if stdout is not connected to a tty.
